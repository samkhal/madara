42  }
43

45  std::vector<std::string>::const_iterator pivot = pivot_list_.begin ();

48

50  {
51  if (*pivot == "{")
52  {

54  }
55  else if (*pivot == "}")
56  {

59  }

64  "KARL COMPILE ERROR: Array name has "

67 
68  exit (-1);
69  }
70  else if (num_closes > num_opens)
71  {
72  madara_logger_ptr_log (logger_, logger::LOG_EMERGENCY,
73  "KARL COMPILE ERROR: Array name has "
74  "more closing braces than opening in %s\n",
75  key.c_str ());
76 
77  exit (-1);
78  }
79  }

81  // this will save us lots of clock cycles each variable access or

95 madara::expression::CompositeArrayReference::expand_key (void) const
96 {
97  if (key_expansion_necessary_)
98  {
99  madara_logger_ptr_log (logger_, logger::LOG_DETAILED,
100  "Variable %s requires variable expansion\n",
101  key_.c_str ());
102 
103  unsigned int count = 0;
104 
105  // add the first token into a string builder
106  std::stringstream builder;
107  std::vector< std::string>::const_iterator token = tokens_.begin ();
108  builder << *token;
109 
110  // move the token to the next in the list.
111  for (++token, ++count; token != tokens_.end (); ++token, ++count)
112  {
113  if (*token != "")
114  {
115  // is the current token a variable to lookup?
116  if (count < pivot_list_.size ()
117  && pivot_list_[count] == "}")
118  {
119  builder << context_.get_record (*token)->to_string ();
120  }
121  else
122  {
123  builder << *token;
124  }
125  }
126  }
127 
128  return builder.str ();
129  }
130  // if there was no need to expand the key, just return
131  // the key
132  else
133  return key_;
134 }

199 madara::expression::CompositeArrayReference::dec (
200  const madara::knowledge::KnowledgeUpdateSettings & settings)
201 {
202  size_t index = size_t (right_->evaluate (settings).to_integer ());
203 
204  if (record_)
205  {
206  // notice that we assume the context is locked
207  // check if we have the appropriate write quality
208  if (!settings.always_overwrite && record_->write_quality < record_->quality)
209  return context_.retrieve_index (expand_key (), index, settings);
210 
211  // cheaper to read than write, so check to see if
212  // we actually need to update quality and status
213  if (record_->write_quality != record_->quality)
214  record_->quality = record_->write_quality;
215 
216  knowledge::KnowledgeRecord result (record_->dec_index (index));
217 
218  context_.mark_and_signal (key_.c_str(), record_);
219 
220  return result;
221  }
222  else
223  {
224  knowledge::KnowledgeRecord result =
225  context_.retrieve_index (expand_key (), index, settings) - knowledge::KnowledgeRecord::Integer (1);
226 
227  if (result.type () == knowledge::KnowledgeRecord::INTEGER)
228  context_.set_index (expand_key (), index, result.to_integer (), settings);
229  else if (result.type () == knowledge::KnowledgeRecord::DOUBLE)
