633 inline int
634 madara::knowledge::KnowledgeBaseImpl::send_modifieds (
635  const std::string & prefix,
636  const EvalSettings & settings)
637 {
638  int result = 0;
639 
640  MADARA_GUARD_TYPE guard (map_.mutex_);
641 
642  if (transports_.size () > 0 && !settings.delay_sending_modifieds)
643  {
644  const knowledge::KnowledgeRecords & modified = map_.get_modifieds ();
645 
646  if (modified.size () > 0)
647  {
648  // if there is not an allowed send_list list
649  if (settings.send_list.size () == 0)
650  {
651  transports_.lock ();
652  // send across each transport
653  for (unsigned int i = 0; i < transports_.size (); ++i, ++result)
654  transports_[i]->send_data (modified);
655 
656  transports_.unlock ();
657 
658  // reset the modified map
659  map_.reset_modified ();
660  }
661  // if there is a send_list
662  else
663  {
664  knowledge::KnowledgeRecords allowed_modifieds;
665  // otherwise, we are only allowed to send a subset of modifieds
666  for (KnowledgeRecords::const_iterator i = modified.begin ();
667  i != modified.end (); ++i)
668  {
669  if (settings.send_list.find (i->first) != settings.send_list.end ())
670  {
671  allowed_modifieds[i->first] = i->second;
672  }
673  }
674 
675  // if the subset was greater than zero, we send the subset
676  if (allowed_modifieds.size () > 0)
677  {
678  transports_.lock ();
679 
680  // send across each transport
681  for (unsigned int i = 0; i < transports_.size (); ++i, ++result)
682  transports_[i]->send_data (allowed_modifieds);
683 
684  transports_.unlock ();
685 
686  // reset modified list for the allowed modifications
687  for (KnowledgeRecords::const_iterator i = allowed_modifieds.begin ();
688  i != allowed_modifieds.end (); ++i)
689  {
690  map_.reset_modified (i->first);
691  }
692  }
693  }
694 
695  map_.inc_clock (settings);
696 
697  if (settings.signal_changes)
698  map_.signal (false);
699  }
700  else
701  {
702  madara_logger_log (map_.get_logger (), logger::LOG_DETAILED,
703  "%s: no modifications to send\n", prefix.c_str ());
704 
705  result = -1;
706  }
707  }
708  else
709  {
710  if (transports_.size () == 0)
711  {
712  madara_logger_log (map_.get_logger (), logger::LOG_DETAILED,
713  "%s: no transport configured\n", prefix.c_str ());
714 
715  result = -2;
716  }
717  else if (settings.delay_sending_modifieds)
718  {
719  madara_logger_log (map_.get_logger (), logger::LOG_DETAILED,
720  "%s: user requested to not send modifieds\n", prefix.c_str ());
721 
722  result = -3;
723  }
724  }
725 
726  return result;
727 }
