671 madara::knowledge::KnowledgeRecord
672 madara::knowledge::KnowledgeBaseImpl::wait (
673 CompiledExpression & ce,
674 const WaitSettings & settings)
675 {
676  // get current time of day
677  ACE_Time_Value current = ACE_High_Res_Timer::gettimeofday ();
678  ACE_Time_Value max_wait, sleep_time, next_epoch;
679  ACE_Time_Value poll_frequency, last = current;
680 
681  if (settings.poll_frequency >= 0)
682  {
683  max_wait.set (settings.max_wait_time);
684  max_wait = current + max_wait;
685 
686  poll_frequency.set (settings.poll_frequency);
687  next_epoch = current + poll_frequency;
688  }
689 
690  // print the post statement at highest log level (cannot be masked)
691  if (settings.pre_print_statement != "")
692  map_.print (settings.pre_print_statement, logger::LOG_EMERGENCY);
693 
694  // lock the context
695  map_.lock ();
696 
697  madara_logger_log (map_.get_logger (), logger::LOG_MAJOR,
698  "KnowledgeBaseImpl::wait:" \
699  " waiting on %s\n", ce.logic.c_str ());
700 
701  madara::knowledge::KnowledgeRecord last_value = ce.expression.evaluate (settings);
702 
703  madara_logger_log (map_.get_logger (), logger::LOG_DETAILED,
704  "KnowledgeBaseImpl::wait:" \
705  " completed first eval to get %s\n",
706  last_value.to_string ().c_str ());
707 
708  send_modifieds ("KnowledgeBaseImpl:wait", settings);
709 
710  map_.unlock ();
711 
712  current = ACE_High_Res_Timer::gettimeofday ();
713 
714  // wait for expression to be true
715  while (!last_value.to_integer () &&
716  (settings.max_wait_time < 0 || current < max_wait))
717  {
718  madara_logger_log (map_.get_logger (), logger::LOG_DETAILED,
719  "KnowledgeBaseImpl::wait:" \
720  " current is %" PRIu64 ".%" PRIu64 " and max is %" PRIu64 ".%" PRIu64 " (poll freq is %f)\n",
721  current.sec (), current.usec (), max_wait.sec (), max_wait.usec (),
722  settings.poll_frequency);
723 
724  madara_logger_log (map_.get_logger (), logger::LOG_DETAILED,
725  "KnowledgeBaseImpl::wait:" \
726  " last value didn't result in success\n");
727 
728  // Unlike the other wait statements, we allow for a time based wait.
729  // To do this, we allow a user to specify a
730  if (settings.poll_frequency > 0)
731  {
732  if (current < next_epoch)
733  {
734  sleep_time = next_epoch - current;
735  madara::utility::sleep (sleep_time);
736  }
737 
738  next_epoch = next_epoch + poll_frequency;
739  }
740  else
741  map_.wait_for_change (true);
742 
743  // relock - basically we need to evaluate the tree again, and
744  // we can't have a bunch of people changing the variables as
745  // while we're evaluating the tree.
746  map_.lock ();
747 
748 
749  madara_logger_log (map_.get_logger (), logger::LOG_MAJOR,
750  "KnowledgeBaseImpl::wait:" \
751  " waiting on %s\n", ce.logic.c_str ());
752 
753  last_value = ce.expression.evaluate (settings);
754 
755  madara_logger_log (map_.get_logger (), logger::LOG_DETAILED,
756  "KnowledgeBaseImpl::wait:" \
757  " completed eval to get %s\n",
758  last_value.to_string ().c_str ());
759 
760  send_modifieds ("KnowledgeBaseImpl:wait", settings);
761 
762  map_.unlock ();
763  map_.signal ();
764 
765  // get current time
766  current = ACE_High_Res_Timer::gettimeofday ();
767 
768  } // end while (!last)
769 
770  if (current >= max_wait)
771  {
772  madara_logger_log (map_.get_logger (), logger::LOG_MAJOR,
773  "KnowledgeBaseImpl::wait:" \
774  " Evaluate did not succeed. Timeout occurred\n");
775  }
776 
777  // print the post statement at highest log level (cannot be masked)
778  if (settings.post_print_statement != "")
779  map_.print (settings.post_print_statement, logger::LOG_ALWAYS);
780 
781  return last_value;
782 }
