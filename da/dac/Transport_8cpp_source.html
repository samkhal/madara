784  }

803  }

810 long madara::transport::Base::prep_send (
811  const madara::knowledge::KnowledgeRecords & orig_updates,
812  const char * print_prefix)
813 {
814  // check to see if we are shutting down
815  long ret = this->check_transport ();
816  if (-1 == ret)
817  {
818  madara_logger_log (context_.get_logger (), logger::LOG_MAJOR,
819  "%s: transport has been told to shutdown",
820  print_prefix);
821 
822  return ret;
823  }
824  else if (-2 == ret)
825  {
826  madara_logger_log (context_.get_logger (), logger::LOG_MAJOR,
827  "%s: transport is not valid",
828  print_prefix);
829 
830  return ret;
831  }
832 
833  // get the maximum quality from the updates
834  uint32_t quality = knowledge::max_quality (orig_updates);
835  bool reduced = false;
836 
837  knowledge::KnowledgeMap filtered_updates;
838 
839  madara_logger_log (context_.get_logger (), logger::LOG_MINOR,
840  "%s:" \
841  " Applying filters before sending...\n",
842  print_prefix);
843 
844  TransportContext transport_context (TransportContext::SENDING_OPERATION,
845  receive_monitor_.get_bytes_per_second (),
846  send_monitor_.get_bytes_per_second (),
847  (uint64_t) time (NULL), (uint64_t) time (NULL),
848  settings_.write_domain,
849  id_);
850 
851  bool dropped = false;
852 
853  if (send_monitor_.is_bandwidth_violated (
854  settings_.get_send_bandwidth_limit ()))
855  {
856  dropped = true;
857  madara_logger_log (context_.get_logger (), logger::LOG_MAJOR,
858  "%s:" \
859  " Send monitor has detected violation of bandwidth limit." \
860  " Dropping packet...\n", print_prefix);
861  }
862  else if (receive_monitor_.is_bandwidth_violated (
863  settings_.get_total_bandwidth_limit ()))
864  {
865  dropped = true;
866  madara_logger_log (context_.get_logger (), logger::LOG_MAJOR,
867  "%s:" \
868  " Receive monitor has detected violation of bandwidth limit." \
869  " Dropping packet...\n", print_prefix);
870  }
871 
872  if (!dropped && packet_scheduler_.add ())
873  {
878  for (knowledge::KnowledgeRecords::const_iterator i = orig_updates.begin ();
879  i != orig_updates.end (); ++i)
880  {
881  madara_logger_log (context_.get_logger (), logger::LOG_MAJOR,
882  "%s:" \
883  " Calling filter chain.\n", print_prefix);
884 
885  // filter the record according to the send filter chain
886  knowledge::KnowledgeRecord result = settings_.filter_send (*i->second, i->first,
887  transport_context);
888 
889  madara_logger_log (context_.get_logger (), logger::LOG_MAJOR,
890  "%s:" \
891  " Filter returned.\n", print_prefix);
892 
893  if (result.exists ())
894  {
895  madara_logger_log (context_.get_logger (), logger::LOG_MINOR,
896  "%s:" \
897  " Adding record to update list.\n", print_prefix);
898 
899  filtered_updates[i->first] = result;
900  }
901  else
902  {
903  madara_logger_log (context_.get_logger (), logger::LOG_MAJOR,
904  "%s:" \
905  " Filter removed record from update list.\n", print_prefix);
906  }
907  }
908 
909  const knowledge::KnowledgeMap & additionals = transport_context.get_records ();
910 
911  for (knowledge::KnowledgeMap::const_iterator i = additionals.begin ();
912  i != additionals.end (); ++i)
913  {
914  madara_logger_log (context_.get_logger (), logger::LOG_MAJOR,
915  "%s:" \
916  " Filter added a record %s to the update list.\n",
917  print_prefix, i->first.c_str ());
918  filtered_updates[i->first] = i->second;
919  }
920  }
921  else
922  {
923  madara_logger_log (context_.get_logger (), logger::LOG_MAJOR,
924  "%s:" \
925  " Packet scheduler has dropped packet...\n", print_prefix);
926 
927  return 0;
928  }
929 
930  madara_logger_log (context_.get_logger (), logger::LOG_MINOR,
931  "%s:" \
932  " Applying %d aggregate update send filters to %d updates...\n",
933  print_prefix, (int)settings_.get_number_of_send_aggregate_filters (),
934  (int)filtered_updates.size ());
935 
936  // apply the aggregate filters
937  if (settings_.get_number_of_send_aggregate_filters () > 0 &&
938  filtered_updates.size () > 0)
939  {
940  settings_.filter_send (filtered_updates, transport_context);
941  }
942  else
943  {
944  madara_logger_log (context_.get_logger (), logger::LOG_MINOR,
945  "%s:" \
946  " No aggregate send filters were applied...\n",
947  print_prefix);
948  }
949 
950  packet_scheduler_.print_status (logger::LOG_DETAILED, print_prefix);
951 
952  madara_logger_log (context_.get_logger (), logger::LOG_MINOR,
953  "%s:" \
954  " Finished applying filters before sending...\n",
955  print_prefix);
956 
957  if (filtered_updates.size () == 0)
958  {
959  madara_logger_log (context_.get_logger (), logger::LOG_MINOR,
960  "%s:" \
961  " Filters removed all data. Nothing to send.\n",
962  print_prefix);
963 
964  return 0;
965  }
966 
967  // allocate a buffer to send
968  char * buffer = buffer_.get_ptr ();
969  int64_t buffer_remaining = settings_.queue_length;
970 
971  if (buffer == 0)
972  {
973  madara_logger_log (context_.get_logger (), logger::LOG_EMERGENCY,
974  "%s:" \
975  " Unable to allocate buffer of size " PRIu32 ". Exiting thread.\n",
976  print_prefix,
977  settings_.queue_length);
978 
979  return -3;
980  }
981 
982 
983  // set the header to the beginning of the buffer
984  MessageHeader * header = 0;
985 
986  if (settings_.send_reduced_message_header)
987  {
988  madara_logger_log (context_.get_logger (), logger::LOG_MINOR,
989  "%s:" \
990  " Preparing message with reduced message header.\n",
991  print_prefix);
992 
993  header = new ReducedMessageHeader ();
994  reduced = true;
995  }
996  else
997  {
998  madara_logger_log (context_.get_logger (), logger::LOG_MINOR,
999  "%s:" \
1000  " Preparing message with normal message header.\n",
1001  print_prefix);
1002 
1003  header = new MessageHeader ();
1004  }
1005 
1006  // get the clock
1007  header->clock = context_.get_clock ();
1008 
1009  if (!reduced)
1010  {
1011  // copy the domain from settings
1012  strncpy (header->domain, this->settings_.write_domain.c_str (),
1013  sizeof (header->domain) - 1);
1014 
1015  // get the quality of the key
1016  header->quality = quality;
1017 
1018  // copy the message originator (our id)
1019  strncpy (header->originator, id_.c_str (), sizeof (header->originator) - 1);
1020 
1021  // send data is generally an assign type. However, MessageHeader is
1022  // flexible enough to support both, and this will simply our read thread
1023  // handling
1024  header->type = madara::transport::MULTIASSIGN;
1025 
1026  }
1027 
1028  // set the time-to-live
1029  header->ttl = settings_.get_rebroadcast_ttl ();
1030 
1031  header->updates = uint32_t (filtered_updates.size ());
1032 
1033  // compute size of this header
1034  header->size = header->encoded_size ();
1035 
1036  // keep track of the maximum buffer size for encoding
1037  int max_buffer_size = (int)buffer_remaining;
1038 
1039  // set the update to the end of the header
1040  char * update = header->write (buffer, buffer_remaining);
1041  uint64_t * message_size = (uint64_t *)buffer;
1042 
1043  // Message header format
1044  // [size|id|domain|originator|type|updates|quality|clock|list of updates]
1045 
1059  // zero out the memory
1060  //memset(buffer, 0, madara::transport::MAX_PACKET_SIZE);
1061 
1062  // Message update format
1063  // [key|value]
1064 
1065  int j = 0;
1066  for (knowledge::KnowledgeMap::const_iterator i = filtered_updates.begin ();
1067  i != filtered_updates.end (); ++i, ++j)
1068  {
1069  update = i->second.write (update, i->first, buffer_remaining);
1070

1072  {
1073  madara_logger_log (context_.get_logger (), logger::LOG_MINOR,
1074  "%s:" \

1076  print_prefix,
1077  j, i->first.c_str (), i->second.type (), i->second.size ());
1078  }
1079  else
1080  {
1081  madara_logger_log (context_.get_logger (), logger::LOG_EMERGENCY,
1082  "%s:" \
1083  " unable to encode update[%d] => %s of type %"
1084  PRId32 " and size %" PRIu32 "\n",
1085  print_prefix,
1086  j, i->first.c_str (), i->second.type (), i->second.size ());
1087  }
1088  }
1089 
1090  long size (0);
1091 
1092  if (buffer_remaining > 0)
1093  {
1094  size = (long)(settings_.queue_length - buffer_remaining);
1095  *message_size = madara::utility::endian_swap ((uint64_t)size);
1096 
1097  // before we send to others, we first execute rules
1098  if (settings_.on_data_received_logic.length () != 0)
1099  {
1100 #ifndef _MADARA_NO_KARL_
1101 
1102  madara_logger_log (context_.get_logger (), logger::LOG_MAJOR,
1103  "%s:" \
1104  " evaluating rules in %s\n",
1105  print_prefix,
1106  settings_.on_data_received_logic.c_str ());
1107

1109 
1110  madara_logger_log (context_.get_logger (), logger::LOG_MAJOR,
1111  "%s:" \
1112  " rules have been successfully evaluated\n",
1113  print_prefix);
1114 
1115 #endif // _MADARA_NO_KARL_
1116 
1117  }
1118  else
1119  {
1120  madara_logger_log (context_.get_logger (), logger::LOG_MINOR,
1121  "%s:" \
1122  " no permanent rules were set\n",
1123  print_prefix);
1124  }
1125  }
1126 
1127  madara_logger_log (context_.get_logger (), logger::LOG_MAJOR,
1128  "%s:" \
1129  " calling encode filters\n",
1130  print_prefix);
1131 
1132  // buffer is ready encoding
1133  size = (long)settings_.filter_encode ((unsigned char *)buffer_.get_ptr (),
1134  (int)size, max_buffer_size);
1135 
1136  madara_logger_log (context_.get_logger (), logger::LOG_MINOR,
1137  "%s:" \
1138  " header info before encode: %s\n",
1139  print_prefix, header->to_string ().c_str ());
1140 
1141  delete header;
1142 
1143  return size;
