220 // Copy a linked list of nodes
221 template <class T> void
222 madara::utility::LStack<T>::copy_list (
223  const madara::utility::LStack<T> &rhs)
224 {
225  LStack<T> temp;
226  LStackNode<T>* prev = 0;
227 
228  // Iterate along the list of stack nodes in <s>, creating a new
229  // corresponding node and chaining them along. Note that we
230  // can't use push() to insert at the head since it will reverse
231  // the stack.
232 
233  ::std::auto_ptr <LStackNode<T> > new_node;
234 
235  for (typename LStack<T>::const_iterator it = rhs.begin ();
236  it != rhs.end ();
237  ++it)
238  {
239  new_node.reset (new LStackNode<T> (*it));
240 
241  if (it == rhs.begin ())
242  {
243  // special case for the first iteration: set the head element of
244  // temporary stack
245  temp.head_ = new_node.release ();
246  prev = temp.head_;
247  }
248  else
249  {
250  // standard case: add one element to prev
251  prev->next_ = new_node.release ();
252  prev = prev->next_;
253  }
254 
255  // make sure that the element count of temp stays correct
256  ++temp.count_;
257  }
258 
259  // we only swap the lists if the temporary list has been successfully
260  // created. This ensures strong exception guarantees.
261  ::std::swap (head_, temp.head_);
262 
263  // set the counts correctly
264  ::std::swap (count_, temp.count_);
265 }
