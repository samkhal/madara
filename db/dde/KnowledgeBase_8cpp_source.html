1319 madara::knowledge::KnowledgeRecord
1320 madara::knowledge::KnowledgeBase::wait (
1321  CompiledExpression & expression,
1322  const WaitSettings & settings)
1323 {
1324  knowledge::KnowledgeRecord result;
1325 
1326  if (context_)
1327  {
1338  ACE_Time_Value current = ACE_High_Res_Timer::gettimeofday ();
1339  ACE_Time_Value max_wait, sleep_time, next_epoch;
1340  ACE_Time_Value poll_frequency, last = current;
1341 
1342  if (settings.poll_frequency >= 0)
1343  {
1344  max_wait.set (settings.max_wait_time);
1345  max_wait = current + max_wait;
1346 
1347  poll_frequency.set (settings.poll_frequency);
1348  next_epoch = current + poll_frequency;
1349  }
1350 
1351  // print the post statement at highest log level (cannot be masked)
1352  if (settings.pre_print_statement != "")
1353  context_->print (settings.pre_print_statement, logger::LOG_ALWAYS);
1354 
1355  // lock the context
1356  context_->lock ();
1357 
1358  madara_logger_log (context_->get_logger (), logger::LOG_MAJOR,
1359  "KnowledgeBase::wait:" \
1360  " waiting on %s\n", expression.logic.c_str ());
1361 
1362  madara::knowledge::KnowledgeRecord last_value = expression.expression.evaluate (settings);
1363 
1364  madara_logger_log (context_->get_logger (), logger::LOG_DETAILED,
1365  "KnowledgeBase::wait:" \
1366  " completed first eval to get %s\n",
1367  last_value.to_string ().c_str ());
1368 
1369  send_modifieds ("KnowledgeBase:wait", settings);
1370 
1371  context_->unlock ();
1372 
1373  current = ACE_High_Res_Timer::gettimeofday ();
1374 
1375  // wait for expression to be true
1376  while (!last_value.to_integer () &&
1377  (settings.max_wait_time < 0 || current < max_wait))
1378  {
1379  madara_logger_log (context_->get_logger (), logger::LOG_DETAILED,
1380  "KnowledgeBase::wait:" \
1381  " current is %llu.%llu and max is %llu.%llu (poll freq is %f)\n",
1382  (unsigned long long)current.sec (),
1383  (unsigned long long)current.usec (),
1384  (unsigned long long)max_wait.sec (),
1385  (unsigned long long)max_wait.usec (),
1386  settings.poll_frequency);
1387 
1388  madara_logger_log (context_->get_logger (), logger::LOG_DETAILED,
1389  "KnowledgeBase::wait:" \
1390  " last value didn't result in success\n");
1391 
1392  // Unlike the other wait statements, we allow for a time based wait.
1393  // To do this, we allow a user to specify a
1394  if (settings.poll_frequency > 0)
1395  {
1396  if (current < next_epoch)
1397  {
1398  sleep_time = next_epoch - current;
1399  madara::utility::sleep (sleep_time);
1400  }
1401 
1402  next_epoch = next_epoch + poll_frequency;
1403  }
1404  else
1405  context_->wait_for_change (true);
1406 
1407  // relock - basically we need to evaluate the tree again, and
1408  // we can't have a bunch of people changing the variables as
1409  // while we're evaluating the tree.
1410  context_->lock ();
1411 
1412 
1413  madara_logger_log (context_->get_logger (), logger::LOG_MAJOR,
1414  "KnowledgeBase::wait:" \
1415  " waiting on %s\n", expression.logic.c_str ());
1416 
1417  last_value = expression.expression.evaluate (settings);
1418 
1419  madara_logger_log (context_->get_logger (), logger::LOG_DETAILED,
1420  "KnowledgeBase::wait:" \
1421  " completed eval to get %s\n",
1422  last_value.to_string ().c_str ());
1423 
1424  send_modifieds ("KnowledgeBase:wait", settings);
1425 
1426  context_->unlock ();
1427  context_->signal ();
1428 
1429  // get current time
1430  current = ACE_High_Res_Timer::gettimeofday ();
1431 
1432  } // end while (!last)
1433 
1434  if (current >= max_wait)
1435  {
1436  madara_logger_log (context_->get_logger (), logger::LOG_MAJOR,
1437  "KnowledgeBase::wait:" \
1438  " Evaluate did not succeed. Timeout occurred\n");
1439  }
1440 
1441  // print the post statement at highest log level (cannot be masked)
1442  if (settings.post_print_statement != "")
1443  context_->print (settings.post_print_statement, logger::LOG_ALWAYS);
1444 
1445  return last_value;
1446  }
1447  else if (impl_.get ())
1448  {
1449  result = impl_->wait (expression, settings);
1450  }
1451 
1452  return result;
1453 }
